10/**
11 * @brief Call callback function with all containers items combination @n
12 * 引数のコンテナの要素の全組合せを引数ににcallback 関数を呼びます
13 *
14 * @tparam T container type(expect bi-direction iterator) @n
15 * コンテナ型(双方向イテレータを持っていることを期待しています)
16 * @tparam F callback function type.
17 * expect: bool func(const T &v, long long k) @n
18 * コールバック関数型: bool func(const T &v, long long k)
19 * の関数を期待しています
20 * @param v container for listing the combination @n
21 * 組合せを列挙するコンテナ
22 * @param k Number of elements in the combination @n
23 * 組合せの要素数
24 * @param callback The callback function.
25 * expect: bool func(const T &v, long long k) @n
26 * コールバック関数。以下の型を期待しています。
27 * bool func(const T &v, long long k)
28 *
29 * Reorder argument v to list all combination. The first k elements
30 * are target elements. If there are elements with the same value, they are
31 * regarded as the same combination and are not listed. If you list all
32 * combination even if container includes equals elements, please call @link
33 * foreach_combination_all @endlink function.
34 * The callback function is called each combination. If you return @b true in
35 * the callback function, continue listing combination. If you return @b false
36 * in the callback function, abort listing combination. In this case, argument
37 * container may be different order than it was before this function was called.
38 *
39 * 全組合せを列挙するために、引数 v の 要素を並び替えていく。先頭 k 個が組合せ
40 * の要素となっています。同値の要素がある場合は、同じ組合せとみなし列挙はしません。
41 * 重複要素があってもすべての組合せを列挙したい場合は、 @link
42 * foreach_combination_all @endlink 関数を呼び出してください。
43 * コールバック関数がそれぞれの組合せ毎に呼び出されます。もし、コールバック関数から
44 * @b true が返った場合は、列挙を続けます。もし、コールバック関数から @b false
45 * が返った
46 * 場合は、列挙を中断します。この場合、引数のコンテナの並び順は関数の呼び出し前と
47 * 異なる順番になっている場合があります。
48 *
49 * example: normal case
50 * @code{.cpp}
51 * #include "bcpl/combination"
52 * #include <cassert>
53 * #include <vector>
54 *
55 * int main() {
56 * using VI = std::vector<int>;
57 * using VVI = std::vector<VI>;
58 * VI v = {0, 1, 2};
59 * VI original = v;
60 * VVI combinations;
61 *
62 * bcpl::foreach_combination(v, 2, [&combinations](const VI &v, long long k) {
63 * combinations.push_back({v[0], v[1]});
64 * return true;
65 * });
66 *
67 * assert(combinations.size() == static_cast<size_t>(3));
68 * assert(combinations[0] == VI({0, 1}));
69 * assert(combinations[1] == VI({0, 2}));
70 * assert(combinations[2] == VI({1, 2}));
71 * assert(v == original);
72 * return 0;
73 * }
74 * @endcode
75 *
76 * example: same elements
77 * @code{.cpp}
78 * #include "bcpl/combination"
79 * #include <cassert>
80 * #include <vector>
81 *
82 * int main() {
83 * using VI = std::vector<int>;
84 * using VVI = std::vector<VI>;
85 * VI v = {1, 1, 2};
86 * VI original = v;
87 * VVI combinations;
88 *
89 * bcpl::foreach_combination(v, 2, [&combinations](const VI &v, long long k) {
90 * combinations.push_back({v[0], v[1]});
91 * return true;
92 * });
93 *
94 * assert(combinations.size() == static_cast<size_t>(2));
95 * assert(combinations[0] == VI({1, 1}));
96 * assert(combinations[1] == VI({1, 2}));
97 * assert(v == original);
98 * return 0;
99 * }
100 * @endcode
101 *
102 * example: break in the middle
103 * @code{.cpp}
104 * #include "bcpl/combination"
105 * #include <cassert>
106 * #include <vector>
107 *
108 * int main() {
109 * using VI = std::vector<int>;
110 * using VVI = std::vector<VI>;
111 * VI v = {0, 1, 2};
112 * VI original = v;
113 * VVI combinations;
114 *
115 * int i = 0;
116 * bcpl::foreach_combination(v, 2,
117 * [&combinations, &i](const VI &v, long long k) {
118 * combinations.push_back({v[0], v[1]});
119 * ++i;
120 * if (i == 2) {
121 * return false;
122 * }
123 * return true;
124 * });
125 *
126 * assert(combinations.size() == static_cast<size_t>(2));
127 * assert(combinations[0] == VI({0, 1}));
128 * assert(combinations[1] == VI({0, 2}));
129 * assert(v != original);
130 * return 0;
131 * }
132 * @endcode
133 */
134template <typename T, typename F>
135void foreach_combination(T &v, long long k, F &&callback) {
136 if (v.empty() || v.size() < static_cast<std::size_t>(k) || k == 0) {
137 return;
138 }
139 if (!callback(v, k)) {
140 return;
141 }
142
143 const auto first = v.begin();
144 const auto last = v.end();
145 auto subset = std::next(first, k);
146 if (subset == last) {
147 return;
148 }
149 auto src = subset;
150
151 while (first != src) {
152 src--;
153 if (*src < *(last - 1)) {
154 auto dest = subset;
155 while (*src >= *dest) {
156 dest++;
157 }
158 std::iter_swap(src, dest);
159 std::rotate(src + 1, dest + 1, last);
160 std::rotate(subset, subset + (last - dest) - 1, last);
161 if (!callback(v, k)) {
162 return;
163 }
164 }
165 }
166
167 std::rotate(first, subset, last);
168}

170/**
171 * @brief Call callback function with all containers items combination even if
172 * it includes same elements @n
173 * 引数のコンテナの要素が同一の値を含んていてもの全組合せを引数ににcallback
174 * 関数を呼びます
175 *
176 * @tparam T container type(expect bi-direction iterator) @n
177 * コンテナ型(双方向イテレータを持っていることを期待しています)
178 * @tparam F callback function type.
179 * expect: bool func(const T &v) @n
180 * コールバック関数型: bool func(const T &v)
181 * の関数を期待しています
182 * @param v container for listing the combination @n
183 * 組合せを列挙するコンテナ
184 * @param k Number of elements in the combination @n
185 * 組合せの要素数
186 * @param callback The callback function.
187 * expect: bool func(const T &v) @n
188 * コールバック関数。以下の型を期待しています
189 * bool func(const T &v)
190 *
191 * Call the callback function with element set vector which includes only
192 * combination data. The basic usage is almost same as @link foreach_combination
193 * @endlink function but this function list all combination set even if
194 * container includes equal element with some same data set. The callback
195 * function is called each combination. If you return @b true in the callback
196 * function, continue listing combination. If you return @b false in the
197 * callback function, abort listing combination. Even if you abort listing,
198 * original vector order isn't changed.
199 *
200 * 組合せの要素を含む配列を引数にコールバック関数を呼び出す。基本的な使い方は
201 * @link foreach_combination
202 * @endlink
203 * 関数と同じですが、この関数はコンテナの要素に同じ値を含んていても重複した組合せも含め
204 * 全ての組合せを列挙します。
205 * コールバック関数がそれぞれの組合せ毎に呼び出されます。もし、コールバック関数から
206 * @b true が返った場合は、列挙を続けます。もし、コールバック関数から @b false
207 * が返った
208 * 場合は、列挙を中断します。この場合でも、引数のコンテナの並び順に変更はありません。
209 */
210template <typename T, typename F>
211void foreach_combination_all(const T &v, long long k, F &&callback) {
212 using VS = std::vector<size_t>;
213 VS wv(v.size());
214 std::iota(wv.begin(), wv.end(), 0);
215
216 T combi(static_cast<std::size_t>(k));
217 foreach_combination(wv, k,
218 [&v, &combi, &callback](const VS &cv, long long ck) {
219 for (long long i = 0; i < ck; ++i) {
220 combi[static_cast<std::size_t>(i)] =
221 v[cv[static_cast<std::size_t>(i)]];
222 }
223 return callback(combi);
224 });
225}
